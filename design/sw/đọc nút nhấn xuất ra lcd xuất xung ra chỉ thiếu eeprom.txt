/*================= HÀM KHỞI TẠO PWM =================*/
void PWM_SetFrequency(uint32_t freq)
{
    if(freq == 0)
    {
        HAL_TIM_PWM_Stop(&htim2, TIM_CHANNEL_2);
        return;
    }

    uint32_t PSC = 72 - 1;      // 72MHz / 72 = 1MHz
    uint32_t ARR = 1000000 / freq;

    htim2.Instance->PSC  = PSC;
    htim2.Instance->ARR  = ARR;
    htim2.Instance->CCR2 = ARR / 2; // Duty 50%

    HAL_TIM_PWM_Start(&htim2, TIM_CHANNEL_2);
}

/*================= KIỂM TRA NÚT NHẤN =================*/
uint32_t ReadNoteKey(void)
{
    if(HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_8) == GPIO_PIN_RESET) return NOTE_C4;
    if(HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_9) == GPIO_PIN_RESET) return NOTE_D4;
    if(HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_10) == GPIO_PIN_RESET) return NOTE_E4;
    if(HAL_GPIO_ReadPin(GPIOA, GPIO_PIN_11) == GPIO_PIN_RESET) return NOTE_F4;
    if(HAL_GPIO_ReadPin(GPIOB, GPIO_PIN_12) == GPIO_PIN_RESET) return NOTE_G4;
    if(HAL_GPIO_ReadPin(GPIOB, GPIO_PIN_13) == GPIO_PIN_RESET) return NOTE_A4;
    if(HAL_GPIO_ReadPin(GPIOB, GPIO_PIN_14) == GPIO_PIN_RESET) return NOTE_B4;
    if(HAL_GPIO_ReadPin(GPIOB, GPIO_PIN_15) == GPIO_PIN_RESET) return NOTE_C5;

    return 0; // Không nhấn nút nào
}

#define LCD_RS_Pin GPIO_PIN_3
#define LCD_RW_Pin GPIO_PIN_4
#define LCD_E_Pin  GPIO_PIN_5

#define LCD_D5_Pin GPIO_PIN_7
#define LCD_D6_Pin GPIO_PIN_8
#define LCD_D7_Pin GPIO_PIN_9

void LCD_Send4Bits(uint8_t data)
{
  HAL_GPIO_WritePin(GPIOB, LCD_D4_Pin, (data >> 0) & 0x01);
  HAL_GPIO_WritePin(GPIOB, LCD_D5_Pin, (data >> 1) & 0x01);
  HAL_GPIO_WritePin(GPIOB, LCD_D6_Pin, (data >> 2) & 0x01);
  HAL_GPIO_WritePin(GPIOB, LCD_D7_Pin, (data >> 3) & 0x01);
}

void LCD_Enable(void)
{
  HAL_GPIO_WritePin(GPIOB, LCD_E_Pin, GPIO_PIN_SET);
  HAL_Delay(1);
  HAL_GPIO_WritePin(GPIOB, LCD_E_Pin, GPIO_PIN_RESET);
  HAL_Delay(1);
}

void LCD_Command(uint8_t cmd)
{
  HAL_GPIO_WritePin(GPIOB, LCD_RS_Pin, GPIO_PIN_RESET);
  HAL_GPIO_WritePin(GPIOB, LCD_RW_Pin, GPIO_PIN_RESET);
  LCD_Send4Bits(cmd >> 4);
  LCD_Enable();
  LCD_Send4Bits(cmd & 0x0F);
  LCD_Enable();
  HAL_Delay(2);
}

void LCD_Data(uint8_t data)
{
  HAL_GPIO_WritePin(GPIOB, LCD_RS_Pin, GPIO_PIN_SET);
  HAL_GPIO_WritePin(GPIOB, LCD_RW_Pin, GPIO_PIN_RESET);
  LCD_Send4Bits(data >> 4);
  LCD_Enable();
  LCD_Send4Bits(data & 0x0F);
  LCD_Enable();
  HAL_Delay(2);
}

void LCD_Init(void)
{
  HAL_Delay(50);
  LCD_Send4Bits(0x03);
  LCD_Enable();
  HAL_Delay(5);
  LCD_Send4Bits(0x02);
  LCD_Enable();

  LCD_Command(0x28); // 4-bit, 2 dòng, font 5x8
  LCD_Command(0x0C); // Bật hiển thị, tắt con trỏ
  LCD_Command(0x06); // Tự tăng con trỏ
  LCD_Command(0x01); // Xóa màn hình
  HAL_Delay(5);
}

void LCD_Clear(void)
{
  LCD_Command(0x01);
  HAL_Delay(2);
}

void LCD_GotoXY(uint8_t row, uint8_t col)
{
  uint8_t address = (row == 0) ? (0x80 + col) : (0xC0 + col);
  LCD_Command(address);
}

void LCD_Puts(char *str)
{
  while (*str)
  {
    LCD_Data(*str++);
  }
}
